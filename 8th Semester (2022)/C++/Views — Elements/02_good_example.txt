#include <fmt/ranges.h>
#include <ranges>
namespace rg = std::ranges;
namespace vs = rg::views;

int main()
{
	constexpr std::pair<bool, std::string_view> my_data[]
	{
		{ 1, "TheMR"},
		{ 0, "The Great" },
		{ 1, "MR_777" },
		{ 0, "Captain" },
	};

	auto my_range = my_data | vs::filter(&std::remove_all_extents_t<decltype(my_data)>::first);
	fmt::print("{}", my_range | vs::elements<1>);
}
